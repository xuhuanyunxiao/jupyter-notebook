class StatsFeatures(BaseEstimator, TransformerMixin):
    
    def __init__(self):
        self.neg = set()
        f = open("corpus/neg_words.txt","r+", encoding='UTF-8')
        for content in f:
            self.neg.add(content)
        f.close()

    def fit(self, X, y=None):
        return self

    def getcnt(self,x):        
        return len(list(set(x)))

    def getnegcnt(self,x):
        negcnt = 0
        words = x.split()
        for w in words:
            if w in self.neg:
                negcnt = negcnt+1
        return negcnt
    
    def transform(self, X):
        data = []
        for x in X:
            if len(x) == 0:
                length  = 1
            else :
                length = len(x)
            data.append([len(x),self.getcnt(x),self.getcnt(x)/length,
                         self.getnegcnt(x),self.getnegcnt(x)/length])            
        return data

def get_server_res(data, url):
    '''
    
     dict,  DataFrame
    '''
    # data = {'record':[{'id':0,'title':'ss','content':'zzz'},]}
    # data = {"record":marked_human_data.iloc[:5,:3].to_dict(orient = 'records')}
    # url "http://47.93.77.19:10000/correlation_negative"
    headers={'content-type':'application/json'}
    result = requests.post(url,
                      data = json.dumps(data),
                      headers=headers, allow_redirects=True)
    # print(result.text)
    json_data = json.loads(result.text)
    parse_data = []
    for i in range(len(json_data['docs'])):
        parse_data.append([json_data['docs'][i]['id'],
                          json_data['docs'][i]['sec'],
                          json_data['docs'][i]['tendency']])
    parse_data = pd.DataFrame(parse_data, columns = ['id', 'sec', 'tendency'])    
    return parse_data

def get_txt_encode(file_path):
    '''
    Function:
        txt
    Arguments:
        file_path -> string -- 
    Return:
        encode -> string -- txt
    '''
    decode_set = ['utf-8','gb18030','ISO-8859-2','gb2312','gbk', 'ANSI'] # 
    encode = 0
    for k in decode_set:
        try :
            file = open(file_path, 'r', encoding = k).read()
            # print('\n\t %s \n\t %s' %(file_path, k))
            encode = k
            break
        except :
            continue
    if encode:
        return encode
    else :
        return 'No encode'
